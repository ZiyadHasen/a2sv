class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowels = set("aeiou")
        curr = 0
        res = 0

        for i in range(len(s)):
            if s[i] in vowels:
                curr += 1

            if i >= k and s[i - k] in vowels:
                curr -= 1

            res = max(res, curr)

        return res
