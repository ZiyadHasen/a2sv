class Solution:
    def frequencySort(self, s: str) -> str:
        freq = {}
        for c in s:
            freq[c] = freq.get(c, 0) + 1

        chars = sorted(freq.items(), key=lambda x: -x[1])

        res = []
        for c, count in chars:
            res.append(c * count)

        return "".join(res)
