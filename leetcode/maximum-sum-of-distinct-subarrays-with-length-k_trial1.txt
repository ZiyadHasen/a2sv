from typing import List

class Solution:
    def maximumSubarraySum(self, nums: List[int], k: int) -> int:
        freq = {}
        left = 0
        curr = 0
        res = 0

        for right in range(len(nums)):
            curr += nums[right]
            freq[nums[right]] = freq.get(nums[right], 0) + 1

            if right - left + 1 > k:
                freq[nums[left]] -= 1
                if freq[nums[left]] == 0:
                    del freq[nums[left]]
                curr -= nums[left]
                left += 1

            if right - left + 1 == k and len(freq) == k:
                res = max(res, curr)

        return res
