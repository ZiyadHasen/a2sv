class Solution:
    def pancakeSort(self, arr):
        res = []
        n = len(arr)
        for i in range(n, 1, -1):
            m = arr.index(i)
            if m != i - 1:
                if m != 0:
                    res.append(m + 1)
                    arr[:m+1] = arr[:m+1][::-1]
                res.append(i)
                arr[:i] = arr[:i][::-1]
        return res
