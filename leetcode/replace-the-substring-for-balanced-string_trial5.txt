import collections

class Solution:
    def balancedString(self, s: str) -> int:
        count = collections.Counter(s)
        n = len(s)
        k = n // 4
        
        if all(count[char] <= k for char in 'QWER'):
            return 0
        
        res = n
        left = 0
        for right, char in enumerate(s):
            count[char] -= 1
            while left < n and all(count[c] <= k for c in 'QWER'):
                res = min(res, right - left + 1)
                count[s[left]] += 1
                left += 1
                
        return res
