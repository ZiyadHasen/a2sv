class Solution:
    def getRow(self, rowIndex: int) -> List[int]:
        if rowIndex == 0:
            return [1]

        prev= self.getRow(rowIndex-1)
        temp=[1]
        for i in range (len(prev)-1):
            temp.append(prev[i]+prev[i+1])
        temp.append(1) 
        return temp
