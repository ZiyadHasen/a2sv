from typing import List

class Solution:
    def relativeSortArray(self, arr1: List[int], arr2: List[int]) -> List[int]:
        count = {}
        for n in arr1:
            count[n] = count.get(n, 0) + 1

        res = []
        for n in arr2:
            if n in count:
                res.extend([n] * count[n])
                del count[n]

        rest = []
        for n, c in count.items():
            rest.extend([n] * c)

        rest.sort()
        return res + rest
